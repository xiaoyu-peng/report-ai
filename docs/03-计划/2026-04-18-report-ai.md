# ReportAI 智能报告写作平台 实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** 构建一个完整的智能报告写作平台，用户上传参考报告后AI学习其风格，并基于知识库生成新报告。

**Architecture:** 复用第一站 Vue3+SpringBoot 多模块架构，新增 knowledge（文档解析）和 report（AI生成）两个模块，使用 Apache Tika 解析文档，MySQL 全文检索做 RAG，Claude API 负责风格学习和报告生成（SSE流式输出）。

**Tech Stack:** Java 17 / Spring Boot 3.2.5 / MyBatis Plus / MySQL 8.0 / Redis / Vue 3 / Element Plus / Anthropic Claude API / Apache Tika / Docker Compose

package com.reportai.hub.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.cors.CorsConfiguration;
import org.springframework.web.cors.UrlBasedCorsConfigurationSource;
import org.springframework.web.filter.CorsFilter;

@Configuration
public class CorsConfig {
    @Bean
    public CorsFilter corsFilter() {
        CorsConfiguration config = new CorsConfiguration();
        config.addAllowedOriginPattern("*");
        config.addAllowedHeader("*");
        config.addAllowedMethod("*");
        config.setAllowCredentials(true);
        UrlBasedCorsConfigurationSource source = new UrlBasedCorsConfigurationSource();
        source.registerCorsConfiguration("/**", config);
        return new CorsFilter(source);
    }
}

package com.reportai.hub.config;

import com.baomidou.mybatisplus.core.handlers.MetaObjectHandler;
import com.reportai.hub.common.context.UserContext;
import org.apache.ibatis.reflection.MetaObject;
import org.springframework.stereotype.Component;
import java.time.LocalDateTime;

@Component
public class MyMetaObjectHandler implements MetaObjectHandler {
    @Override
    public void insertFill(MetaObject metaObject) {
        this.strictInsertFill(metaObject, "createdAt", LocalDateTime.class, LocalDateTime.now());
        this.strictInsertFill(metaObject, "updatedAt", LocalDateTime.class, LocalDateTime.now());
        this.strictInsertFill(metaObject, "deleted", Integer.class, 0);
        Long userId = UserContext.getUserId();
        if (userId != null) {
            this.strictInsertFill(metaObject, "createdBy", Long.class, userId);
        }
    }

    @Override
    public void updateFill(MetaObject metaObject) {
        this.strictUpdateFill(metaObject, "updatedAt", LocalDateTime.class, LocalDateTime.now());
    }
}

package com.reportai.hub.knowledge.service;

import lombok.extern.slf4j.Slf4j;
import org.apache.tika.Tika;
import org.apache.tika.metadata.Metadata;
import org.apache.tika.parser.AutoDetectParser;
import org.apache.tika.sax.BodyContentHandler;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;

import java.io.InputStream;
import java.util.ArrayList;
import java.util.List;

@Slf4j
@Service
public class DocumentParseService {

    private static final int CHUNK_SIZE = 800;
    private static final int CHUNK_OVERLAP = 100;

    public String parseDocument(MultipartFile file) {
        try (InputStream is = file.getInputStream()) {
            BodyContentHandler handler = new BodyContentHandler(10 * 1024 * 1024);
            Metadata metadata = new Metadata();
            metadata.set(Metadata.CONTENT_TYPE, file.getContentType());
            AutoDetectParser parser = new AutoDetectParser();
            parser.parse(is, handler, metadata);
            return handler.toString().trim();
        } catch (Exception e) {
            log.error("文档解析失败: {}", file.getOriginalFilename(), e);
            throw new RuntimeException("文档解析失败: " + e.getMessage());
        }
    }

    public List<String> splitIntoChunks(String text) {
        List<String> chunks = new ArrayList<>();
        if (text == null || text.trim().isEmpty()) return chunks;

        // 按段落分割
        String[] paragraphs = text.split("\n\n+");
        StringBuilder currentChunk = new StringBuilder();

        for (String paragraph : paragraphs) {
            paragraph = paragraph.trim();
            if (paragraph.isEmpty()) continue;

            if (currentChunk.length() + paragraph.length() > CHUNK_SIZE && currentChunk.length() > 0) {
                chunks.add(currentChunk.toString().trim());
                // 保留重叠
                String overlap = currentChunk.length() > CHUNK_OVERLAP
                        ? currentChunk.substring(currentChunk.length() - CHUNK_OVERLAP)
                        : currentChunk.toString();
                currentChunk = new StringBuilder(overlap).append("\n");
            }
            currentChunk.append(paragraph).append("\n\n");
        }

        if (currentChunk.length() > 0) {
            chunks.add(currentChunk.toString().trim());
        }

        return chunks;
    }
}

                .orderByDesc(KnowledgeBase::getCreatedAt));

    @Override

        return kbMapper.selectById(id);
    }

    @Override
    public void deleteKnowledgeBase(Long id) {
        chunkMapper.delete(new LambdaQueryWrapper<KnowledgeChunk>().eq(KnowledgeChunk::getKbId, id));
        docMapper.delete(new LambdaQueryWrapper<KnowledgeDocument>().eq(KnowledgeDocument::getKbId, id));
        kbMapper.deleteById(id);
    }

    @Override
    public KnowledgeDocument uploadDocument(Long kbId, MultipartFile file) {
        KnowledgeBase kb = kbMapper.selectById(kbId);
        if (kb == null) throw new BusinessException("知识库不存在");

        KnowledgeDocument doc = new KnowledgeDocument();
        doc.setKbId(kbId);
        doc.setFilename(file.getOriginalFilename());
        doc.setFileType(file.getContentType());
        doc.setFileSize(file.getSize());
        doc.setStatus("processing");
        doc.setChunkCount(0);
        docMapper.insert(doc);

        // 异步解析
        processDocumentAsync(doc, file);

        return doc;
    }

    @Async
    public void processDocumentAsync(KnowledgeDocument doc, MultipartFile file) {
        try {
            String content = parseService.parseDocument(file);
            List<String> chunks = parseService.splitIntoChunks(content);

            doc.setContent(content);
            doc.setChunkCount(chunks.size());
            doc.setStatus("ready");
            docMapper.updateById(doc);

            // 保存分块
            for (int i = 0; i < chunks.size(); i++) {
                KnowledgeChunk chunk = new KnowledgeChunk();
                chunk.setDocId(doc.getId());
                chunk.setKbId(doc.getKbId());
                chunk.setContent(chunks.get(i));
                chunk.setChunkIndex(i);
                chunkMapper.insert(chunk);
            }

            // 更新知识库统计
            KnowledgeBase kb = kbMapper.selectById(doc.getKbId());
            if (kb != null) {
                kb.setDocCount(kb.getDocCount() + 1);
                kb.setChunkCount(kb.getChunkCount() + chunks.size());
                kbMapper.updateById(kb);
            }
        } catch (Exception e) {
            log.error("文档处理失败", e);
            doc.setStatus("failed");
            docMapper.updateById(doc);
        }
    }

    @Override
    public List<KnowledgeDocument> listDocuments(Long kbId) {
        return docMapper.selectList(new LambdaQueryWrapper<KnowledgeDocument>()
                .eq(KnowledgeDocument::getKbId, kbId)
                .orderByDesc(KnowledgeDocument::getCreatedAt));
    }

    public void deleteDocument(Long docId) {

        if (doc != null) {

            if (kb != null) {
                kb.setDocCount(Math.max(0, kb.getDocCount() - 1));
                kb.setChunkCount(Math.max(0, kb.getChunkCount() - doc.getChunkCount()));
                kbMapper.updateById(kb);
            }
        }
    }

    @Override
    public List<KnowledgeChunk> searchChunks(Long kbId, String query, int topK) {
        // MySQL 全文检索
        return chunkMapper.fullTextSearch(kbId, query, topK);
    }

package com.reportai.hub.report.service;

import com.fasterxml.jackson.databind.ObjectMapper;
import lombok.extern.slf4j.Slf4j;
import okhttp3.*;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.web.servlet.mvc.method.annotation.SseEmitter;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.List;
import java.util.Map;
import java.util.concurrent.TimeUnit;

@Slf4j
@Service
public class AnthropicService {

    @Value("${anthropic.api-key}")
    private String apiKey;

    private static final String API_URL = "https://api.anthropic.com/v1/messages";
    private static final String MODEL = "claude-sonnet-4-6";
    private final ObjectMapper objectMapper = new ObjectMapper();

    private final OkHttpClient client = new OkHttpClient.Builder()
            .connectTimeout(30, TimeUnit.SECONDS)
            .readTimeout(120, TimeUnit.SECONDS)
            .writeTimeout(30, TimeUnit.SECONDS)
            .build();

    /**
     * 分析文档风格，返回风格描述
     */
    public String analyzeStyle(String documentContent) {
        String prompt = """
                请分析以下文档的写作风格，包括：
                1. 文章结构（章节划分方式）
                2. 语言风格（正式/非正式、专业程度）
                3. 段落特点（长度、过渡方式）
                4. 表达习惯（用词偏好、句式特点）
                5. 格式规范（标题样式、重点标注方式）
                
                请用200字以内简洁描述这些风格特征，便于AI写作时参考。
                
                文档内容：
                """ + documentContent.substring(0, Math.min(3000, documentContent.length()));

        return callClaude(prompt);
    }

    /**
     * 流式生成报告，通过SSE推送到前端
     */
    public void generateReportStream(String systemPrompt, String userPrompt, SseEmitter emitter) {
        Map<String, Object> requestBody = Map.of(
                "model", MODEL,
                "max_tokens", 4096,
                "stream", true,
                "system", systemPrompt,
                "messages", List.of(Map.of("role", "user", "content", userPrompt))
        );

        try {
            String json = objectMapper.writeValueAsString(requestBody);
            Request request = new Request.Builder()
                    .url(API_URL)
                    .post(RequestBody.create(json, MediaType.parse("application/json")))
                    .header("x-api-key", apiKey)
                    .header("anthropic-version", "2023-06-01")
                    .header("content-type", "application/json")
                    .build();

            try (Response response = client.newCall(request).execute()) {
                if (!response.isSuccessful() || response.body() == null) {
                    emitter.send(SseEmitter.event().name("error").data("API调用失败: " + response.code()));
                    emitter.complete();
                    return;
                }

                BufferedReader reader = new BufferedReader(new InputStreamReader(response.body().byteStream()));
                String line;
                while ((line = reader.readLine()) != null) {
                    if (line.startsWith("data: ")) {
                        String data = line.substring(6).trim();
                        if ("[DONE]".equals(data)) {
                            emitter.send(SseEmitter.event().name("done").data("[DONE]"));
                            break;
                        }
                        try {
                            Map<?, ?> event = objectMapper.readValue(data, Map.class);
                            String type = (String) event.get("type");
                            if ("content_block_delta".equals(type)) {
                                Map<?, ?> delta = (Map<?, ?>) event.get("delta");
                                if (delta != null && "text_delta".equals(delta.get("type"))) {
                                    String text = (String) delta.get("text");
                                    if (text != null && !text.isEmpty()) {
                                        emitter.send(SseEmitter.event().name("text").data(text));
                                    }
                                }
                            }
                        } catch (Exception ignored) {}
                    }
                }
                emitter.complete();
            }
        } catch (Exception e) {
            log.error("流式生成失败", e);
            try { emitter.send(SseEmitter.event().name("error").data(e.getMessage())); } catch (Exception ignored) {}
            emitter.completeWithError(e);
        }
    }

    private String callClaude(String prompt) {
        Map<String, Object> requestBody = Map.of(
                "model", MODEL,
                "max_tokens", 1024,
                "messages", List.of(Map.of("role", "user", "content", prompt))
        );

        try {
            String json = objectMapper.writeValueAsString(requestBody);
            Request request = new Request.Builder()
                    .url(API_URL)

package com.reportai.hub.report.service.impl;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;

public class ReportServiceImpl extends ServiceImpl<ReportMapper, Report> implements ReportService {

    @Override
    public Report createDraft(String title, String topic, String keyPoints, Long kbId, Long templateId) {
        Report report = new Report();
        report.setTitle(title);
        report.setTopic(topic);
        report.setKeyPoints(keyPoints);
        report.setKbId(kbId);
        report.setTemplateId(templateId);
        report.setStatus("draft");
        report.setContent("");
        reportMapper.insert(report);
        return report;

    @Override
    public SseEmitter generateReport(Long reportId) {
        SseEmitter emitter = new SseEmitter(180_000L); // 3分钟超时

        Report report = reportMapper.selectById(reportId);
        if (report == null) {
            try { emitter.send(SseEmitter.event().name("error").data("报告不存在")); } catch (Exception ignored) {}
            emitter.complete();
            return emitter;
        }

        // 异步执行生成
        generateAsync(report, emitter);

        return emitter;
    }

    @Async
    public void generateAsync(Report report, SseEmitter emitter) {
        try {
            // 1. 获取模板风格
            String styleDesc = "";
            String templateContent = "";
            if (report.getTemplateId() != null) {
                ReportTemplate template = templateMapper.selectById(report.getTemplateId());
                if (template != null) {
                    styleDesc = template.getStyleDescription() != null ? template.getStyleDescription() : "";
                    templateContent = template.getContent() != null ? template.getContent() : "";
                }
            }

            // 2. RAG 检索相关知识
            String ragContext = "";
            if (report.getKbId() != null && report.getTopic() != null) {
                List<KnowledgeChunk> chunks = knowledgeService.searchChunks(
                        report.getKbId(), report.getTopic(), 8);
                if (!chunks.isEmpty()) {
                    ragContext = chunks.stream()
                            .map(KnowledgeChunk::getContent)
                            .collect(Collectors.joining("\n\n---\n\n"));
                }
            }

            // 3. 构建 System Prompt
            String systemPrompt = buildSystemPrompt(styleDesc, templateContent);

            // 4. 构建 User Prompt
            String userPrompt = buildUserPrompt(report, ragContext);

            // 5. 流式生成
            anthropicService.generateReportStream(systemPrompt, userPrompt, emitter);

        } catch (Exception e) {
            log.error("报告生成失败", e);
            try { emitter.send(SseEmitter.event().name("error").data("生成失败: " + e.getMessage())); } catch (Exception ignored) {}
            emitter.completeWithError(e);
        }
    }

    private String buildSystemPrompt(String styleDesc, String templateContent) {
        StringBuilder sb = new StringBuilder();
        sb.append("你是一位专业的报告撰写专家，擅长根据要求生成高质量的分析报告。\n\n");

        if (!styleDesc.isEmpty()) {
            sb.append("【写作风格要求】\n").append(styleDesc).append("\n\n");
        }

        if (!templateContent.isEmpty()) {
            sb.append("【参考模板结构】\n请参考以下结构组织报告：\n")
              .append(templateContent.substring(0, Math.min(1500, templateContent.length())))
              .append("\n\n");
        }

        sb.append("要求：\n");
        sb.append("1. 报告结构清晰，有明确的章节划分（使用 ## 作为章节标题）\n");
        sb.append("2. 内容专业严谨，逻辑连贯\n");
        sb.append("3. 使用 Markdown 格式输出\n");
        sb.append("4. 字数在1500-3000字之间\n");

        return sb.toString();
    }

    private String buildUserPrompt(Report report, String ragContext) {
        StringBuilder sb = new StringBuilder();
        sb.append("请为以下主题生成一份专业报告：\n\n");
        sb.append("【报告标题】").append(report.getTitle()).append("\n\n");
        sb.append("【核心主题】").append(report.getTopic()).append("\n\n");

        if (report.getKeyPoints() != null && !report.getKeyPoints().isEmpty()) {
            sb.append("【重点要覆盖的内容】\n").append(report.getKeyPoints()).append("\n\n");
        }

        if (!ragContext.isEmpty()) {
            sb.append("【参考知识库内容】\n以下是相关背景资料，请合理引用：\n\n")
              .append(ragContext).append("\n\n");
        }

        sb.append("请直接开始生成报告内容，不要有任何前置说明。");

        return sb.toString();
    }

    @Override
    public Report updateReport(Long id, String title, String content) {
        Report report = reportMapper.selectById(id);
        if (report == null) return null;
        if (title != null) report.setTitle(title);
        if (content != null) {
            report.setContent(content);
            report.setStatus("completed");
            report.setWordCount(content.length());
        }
        reportMapper.updateById(report);
        return report;
    }

    @Override
    public void deleteReport(Long id) {
        reportMapper.deleteById(id);
    }
}

async function handleGenerate() {
  if (!form.value.title || !form.value.topic) return

  generating.value = true
  content.value = ''

  try {
    // 先创建草稿获取ID
    const createRes = await createReport({
      title: form.value.title,
      topic: form.value.topic,
      keyPoints: form.value.keyPoints,
      kbId: form.value.kbId,
      templateId: form.value.templateId
    })
    const reportId = createRes.data?.data?.id
    currentReportId.value = reportId

    // 建立 SSE 连接
    const token = userStore.token || localStorage.getItem('token')
    const evtSource = new EventSource(
      `/api/v1/reports/${reportId}/generate`,
    )

    // 通过Header传Token需要用fetch+ReadableStream
    const response = await fetch(`/api/v1/reports/${reportId}/generate`, {
      headers: { 'Authorization': `Bearer ${token}` }
    })

    const reader = response.body!.getReader()
    const decoder = new TextDecoder()

    while (true) {
      const { done, value } = await reader.read()
      if (done) break

      const text = decoder.decode(value)
      const lines = text.split('\n')

      for (const line of lines) {
        if (line.startsWith('data:')) {
          const data = line.slice(5).trim()
          if (data === '[DONE]') {
            generating.value = false
            ElMessage.success('报告生成完成')
            break
          }
          if (line.startsWith('event:text')) continue
          content.value += data
        }
      }
    }
  } catch (e) {
    ElMessage.error('生成失败，请重试')
  } finally {
    generating.value = false
  }
}

async function saveReport() {
  if (!currentReportId.value) return
  const { updateReport } = await import('@/api/report')
  await updateReport(currentReportId.value, {
    title: form.value.title,
    content: content.value
  })
  ElMessage.success('保存成功')
}

  navigator.clipboard.writeText(content.value)
  ElMessage.success('已复制到剪贴板')
